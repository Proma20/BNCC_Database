CREATE SEQUENCE Account_num   
INCREMENT BY 1
START WITH 9578426485
MAXVALUE   9999999999
NOCACHE
NOCYCLE;



CREATE TABLE ACCOUNT
(
ACCOUNT_NUM number(15),
ACCOUNT_TYPE VARCHAR2(15),
ACCOUNT_OFFICE VARCHAR2(15),
ACCOUNT_PURPOSE VARCHAR2(15),
ACCOUNT_IN_AMOUNT number(15),
ACCOUNT_OUT_AMOUNT number(15),
ACCOUNT_SOURCE number(15),
ACCOUNT_DEST_ACC number(15),
BALANCE number(15),
ACCOUNT_DOT DATE,
ACCOUNT_DETAILS VARCHAR2(15),
CONSTRAINT ACCOUNT_ACCOUNT_NUM_pk PRIMARY KEY(ACCOUNT_NUM),
CONSTRAINT ACCOUNT_BALANCE_CK CHECK(BALANCE>0)
);
